// ============================================================ // FILE: supabase/functions/learn/index.ts // TYPE: Edge Function (deployed) - cron job // DEPLOY: supabase functions deploy learn --no-verify-jwt // ============================================================ // LEARN — the self-improvement rollup. Reads every graded signal and recomputes, // per slice (book / sport / market / edge-bucket / time-to-kickoff), the realized // hit-rate, CLV, beat-close rate and ROI — then writes a recommended `weight` the // scanner uses to prioritize the slices that actually produce CLV. // Deploy: supabase functions deploy learn --no-verify-jwt import { db, authorized, json } from "../_shared/db.ts"; const clamp = (x: number, lo: number, hi: number) => Math.max(lo, Math.min(hi, x)); function edgeBucket(e: number) { if (e == null) return "na"; const p = e * 100; return p < 0 ? "<0%" : p < 2 ? "0-2%" : p < 4 ? "2-4%" : p < 7 ? "4-7%" : "7%+"; } function ttkBucket(commence: string, firstSeen: string) { const h = (new Date(commence).getTime() - new Date(firstSeen).getTime()) / 3.6e6; return h < 2 ? "<2h" : h < 12 ? "2-12h" : h < 48 ? "12-48h" : "48h+"; } interface Agg { n: number; w: number; l: number; push: number; clv: number; beat: number; roi: number; predEdge: number; } const blank = (): Agg => ({ n: 0, w: 0, l: 0, push: 0, clv: 0, beat: 0, roi: 0, predEdge: 0 }); Deno.serve(async (req) => { if (!authorized(req)) return json({ error: "unauthorized" }, 401); // page through graded signals const cols = "best_dec,first_edge,clv,beat_close,result,best_book,sport_key,market,commence_time,first_seen_at"; let from = 0; const PAGE = 1000; const rows: any[] = []; while (from < 50000) { const { data, error } = await db.from("signals").select(cols) .not("graded_at", "is", null).range(from, from + PAGE - 1); if (error || !data?.length) break; rows.push(...data); if (data.length < PAGE) break; from += PAGE; } if (!rows.length) return json({ ok: true, graded: 0, note: "nothing graded yet" }); const slices: Record> = { book: {}, sport: {}, market: {}, edge_bucket: {}, ttk_bucket: {} }; const calib: Record = {}; const add = (m: Record, k: string, r: any) => { const a = m[k] ?? (m[k] = blank()); a.n++; if (r.result === "win") a.w++; else if (r.result === "loss") a.l++; else if (r.result === "push") a.push++; a.clv += r.clv ?? 0; if (r.beat_close) a.beat++; a.roi += r.result === "win" ? (r.best_dec - 1) : r.result === "loss" ? -1 : 0; a.predEdge += r.first_edge ?? 0; }; for (const r of rows) { add(slices.book, r.best_book ?? "?", r); add(slices.sport, r.sport_key ?? "?", r); add(slices.market, r.market ?? "?", r); const eb = edgeBucket(r.first_edge); add(slices.edge_bucket, eb, r); add(calib, eb, r); add(slices.ttk_bucket, ttkBucket(r.commence_time, r.first_seen_at), r); } const weightRows: any[] = []; for (const stype in slices) for (const sval in slices[stype]) { const a = slices[stype][sval]; const decided = a.w + a.l; const hit = decided ? a.w / decided : null; const roi = a.n ? a.roi / a.n : 0; // realized ROI per unit at best price const weight = clamp(1 + Math.tanh(roi * 6) * (a.n / (a.n + 50)), 0.2, 2.0); weightRows.push({ slice_type: stype, slice_value: sval, n: a.n, n_graded: decided, hit_rate: hit, avg_clv: a.n ? a.clv / a.n : null, beat_close_rate: a.n ? a.beat / a.n : null, realized_edge: roi, weight, updated_at: new Date().toISOString(), }); } await db.from("model_weights").upsert(weightRows, { onConflict: "slice_type,slice_value" }); const calibRows = Object.entries(calib).map(([k, a]) => { const decided = a.w + a.l; return { edge_bucket: k, n: a.n, predicted_edge: a.n ? a.predEdge / a.n : null, realized_winrate: decided ? a.w / decided : null, realized_clv: a.n ? a.clv / a.n : null, updated_at: new Date().toISOString() }; }); await db.from("calibration").upsert(calibRows, { onConflict: "edge_bucket" }); return json({ ok: true, graded_rows: rows.length, slices: weightRows.length }); });
